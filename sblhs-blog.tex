\NewDocumentCommand{\examplecite}{sO{auto}d()oom}{%
  \par
  \textcolor{cmdcolour}{%
    \texttt{%
      \textbackslash #2cite%
      \IfBooleanT{#1}{*}%
      \IfNoValueF{#4}{[\verbprint{#4}]}%
      \IfNoValueF{#5}{[\verbprint{#5}]}%
      \{#6\}}}\par
  \@afterheading
  \IfNoValueF{#3}{\quad #3.\space\strut}%
  \IfBooleanTF{#1}
    {\printexamplecite*{#2cite}{#4}{#5}{#6}}
    {\printexamplecite{#2cite}{#4}{#5}{#6}}}

          {\csname #2\endcsname*[#3]{#5}}
          {\csname #2\endcsname[#3]{#5}}}}

       {\IfBooleanTF{#1}
          {\csname #2\endcsname*[#3][]{#5}}
          {\csname #2\endcsname[#3][]{#5}}}

          {\csname #2\endcsname*[#3][#4]{#5}}
